#include <iostream>
using namespace std;
double myPow(double x, int n) {
    if(n==0)
        return 1.0;
    if(n==1)
        return x;
    if(n%2==0)
        return myPow(x*x,n/2);
    if(n>1)
        return(x*myPow(x,n-1));
    return(1/x)*myPow(x,n+1);
}
int main(){
    double num;
    int power;
    cin>>num;
    cin>>power;
    cout<<myPow(num,power);
    return 0;
}
